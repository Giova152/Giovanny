// ============================================ // CONFIGURATION // ============================================ const ADMIN_EMAIL = "admin@midogiova.com"; const SHEET_NAME = "Soumissions"; // ============================================ // FONCTION PRINCIPALE : Réception du Webhook // ============================================ function doPost(e) { try { const data = JSON.parse(e.postData.contents); const eventType = data.event || data.status; // 1. Déterminer le statut de manière lisible let statutFriendly = "Inconnu"; if (eventType === 'payment.succeeded' || eventType === 'succeeded') statutFriendly = "✅ Payé"; else if (eventType === 'payment.failed' || eventType === 'failed') statutFriendly = "❌ Échoué"; else if (eventType === 'payment.canceled' || eventType === 'canceled') statutFriendly = "🟠 Annulé"; else statutFriendly = "ℹ️ " + eventType; // 2. Extraire les données du formulaire (Metadata) et du paiement const formData = data.metadata || {}; formData.statut = statutFriendly; formData.paymentId = data.id || data.payment_id || 'N/A'; formData.amount = data.amount || '50€'; formData.paymentMethod = data.payment_method || 'Non spécifié'; formData.customerEmail = data.customer?.email || formData.email; // 3. ENREGISTRER DANS LE SHEET (Peu importe le statut) addToSheet(formData); // 4. ENVOYER EMAIL (Seulement si c'est payé) if (statutFriendly === "✅ Payé") { sendAdminEmail(formData); } return ContentService.createTextOutput(JSON.stringify({received: true, status: statutFriendly})) .setMimeType(ContentService.MimeType.JSON); } catch (error) { Logger.log("Erreur globale: " + error.toString()); return ContentService.createTextOutput(JSON.stringify({error: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // ============================================ // AJOUTER LA LIGNE DANS GOOGLE SHEETS // ============================================ function addToSheet(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { setupSheet(); // Crée la feuille si elle n'existe pas sheet = ss.getSheetByName(SHEET_NAME); } const row = [ new Date().toLocaleString('fr-FR', { timeZone: 'Africa/Porto-Novo' }), data.statut, data.prenom || '', data.nom || '', data.email || data.customerEmail || '', data.phone || '', data.website || '', data.type_projet || '', data.budget || '', data.delai || '', data.description || '', data.paymentId || '', data.amount || '50€' ]; sheet.appendRow(row); } // ============================================ // ENVOYER L'EMAIL DE NOTIFICATION // ============================================ function sendAdminEmail(data) { const subject = "🎉 Nouveau projet payé : " + (data.prenom || '') + " " + (data.nom || ''); // Note : Le corps HTML utilise les variables collectées const htmlBody = `

Nouveau Dossier Reçu

Client : ${data.prenom} ${data.nom}

Email : ${data.email}

Projet : ${data.type_projet}

Budget : ${data.budget}

Délai : ${data.delai}

Description :
${data.description}

Paiement : ${data.amount} (ID: ${data.paymentId})
`; MailApp.sendEmail({ to: ADMIN_EMAIL, subject: subject, htmlBody: htmlBody }); } // ============================================ // INITIALISATION DES EN-TÊTES (À lancer 1 fois) // ============================================ function setupSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); const headers = [ 'Date', 'Statut', 'Prénom', 'Nom', 'Email', 'Téléphone', 'Site Web', 'Type Projet', 'Budget', 'Délai', 'Description', 'ID Paiement', 'Montant' ]; sheet.getRange(1, 1, 1, headers.length).setValues([headers]) .setBackground('#356646').setFontColor('#ffffff').setFontWeight('bold'); sheet.setFrozenRows(1); }
